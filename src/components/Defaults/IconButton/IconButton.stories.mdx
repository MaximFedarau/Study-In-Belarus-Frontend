# Icon Button

Icon Button is like regular button, but it's `children` prop is of type ReactNode.

## Usage

```tsx
//Types
import { ReactElement } from 'react';

//Components
import PostContent from 'components/Post/PostContent/PostContent.component';

import { Reactions } from 'components/Defaults/Container/Container.component';
import IconButton from 'components/Defaults/IconButton/IconButton.component';

//Icons (using MUI icons)
import ThumbUpIcon from '@mui/icons-material/ThumbUp';
import ThumbDownIcon from '@mui/icons-material/ThumbDown';

export default function Post(): ReactElement {
  function handleLike() {
    //...some actions
  }

  function handleDislike() {
    //...some actions
  }

  return (
    <>
      <PostContent />
      <Reactions>
        <IconButton onClickHandler={handleLike}>
          <ThumbUpIcon />
        </IconButton>
        <IconButton onClickHandler={handleDislike}>
          <ThumbDownIcon />
        </IconButton>
      </Reactions>
    </>
  );
}
```
