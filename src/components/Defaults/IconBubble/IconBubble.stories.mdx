# Icon Bubble

Icon Bubble - a bubble with an icon. But unlike default bubble it is clickable.

## Usage

```tsx
//Types
import { ReactElement } from 'react';

//Components
import IconBubble from 'components/Defaults/IconBubble/IconBubble.component';

//Icons
import ArrowUpwardIcon from '@mui/icons-material/ArrowUpward';

export default function PostContent(): ReactElement {
  function handleBubbleClick(): void {
    //...some actions
  }

  return (
    <div>
      {/*first part of the content */}
      <IconBubble
        size={{ width: '36px', height: '36px' }}
        onClick={handleBubbleClick}
      >
        <ArrowUpwardIcon />
      </IconBubble>
      {/* final part of the content */}
    </div>
  );
}
```
