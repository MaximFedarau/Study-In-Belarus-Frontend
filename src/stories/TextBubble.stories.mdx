# Text Bubble

Text Bubble - is a bubble, which provides more options for working with text.

## Usage

```tsx
//Types
import { ReactElement } from 'react';

//Components
import TextBubble from 'components/Defaults/TextBubble/TextBubble.component';

export default function Example(): ReactElement {
  return (
    <>
      {/*...some content*/}
      <TextBubble size={{ width: '100px', height: '100px' }} text="Super cool!">
        WOW
      </TextBubble>
    </>
  );
}
```
